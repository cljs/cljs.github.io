(defmacro add-data-readers [default-readers]
  (let [data-readers
        (->> (get @env/*compiler* :cljs.analyzer/data-readers)
          (map (fn [[k v]]
                 `['~k (fn [x#] (~(vary-meta (-> v meta :sym) assoc :cljs.analyzer/no-resolve true) x#))]))
          (into {}))]
    `(do (merge ~default-readers ~data-readers))))
