Source code @ clojurescript:src/main/cljs/clojure/zip.cljs
(defn xml-zip
  [root]
    (zipper (complement string?) 
            (comp seq :content)
            (fn [node children]
              (assoc node :content (and children (apply vector children))))
            root))
